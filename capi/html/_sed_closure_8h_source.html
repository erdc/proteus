61  double uSolid[nSpace], //Sediment velocity
62  double nu //Kinematic viscosity
63  )
64  {
65  double du2 = 0.;
66  for (int ii=0; ii<nSpace; ii++)
67  {
68  du2+= (uFluid[ii] - uSolid[ii])*(uFluid[ii] - uSolid[ii]);
69  }
70  double du = sqrt(du2);
71  double weight = 1.;
72 
73 
74  double gDrag1 = (aDarcy_*sedF*nu/((1. - sedF)*grain_*grain_) + betaForch_*du/grain_); // Darcy forchheimer term
75  double Cd = 0.44;
76  double Re = (1-sedF)*du*grain_/nu;
77  double Re_min = 1.0e-3;

80  Cd = 24*(1. + 0.15*pow(fmax(Re_min,Re), 0.687))/fmax(Re_min,Re); //Particle cloud resistance Wen and Yu 1966
81  }
