119 // Computes the product of the corresponding block in the
120 // system matrix (ie. the mass matrix) by 'vect', scale by c_a, and add to 'result'.
121 // NOTE: the 'vect' and 'result' vectors must already have
122 // the size of the total variables&constraints in the system; the procedure
123 // will use the ChVariable offsets (that must be already updated) to know the
124 // indexes in result and vect.
125 void ChVariablesBodyAddedMass::MultiplyAndAdd(ChMatrix<double>& result,
126  const ChMatrix<double>& vect,
127  const double c_a) const {
128  ChMatrixDynamic<> tot = ChMatrixDynamic<>(6, 1);
129  tot.PasteClippedMatrix(vect, this->offset, 0, 6, 1, 0, 0); // for solver warm starting only
130  result.PasteSumMatrix(Mfullmass*tot*c_a, this->offset, 0);
131 }

133 // Add the diagonal of the mass matrix scaled by c_a to 'result'.
134 // NOTE: the 'result' vector must already have the size of system unknowns, ie
135 // the size of the total variables&constraints in the system; the procedure
136 // will use the ChVariable offset (that must be already updated) as index.
137 void ChVariablesBodyAddedMass::DiagonalAdd(ChMatrix<double>& result, const double c_a) const {
138  assert(result.GetColumns() == 1);
139  for (int i = 0; i < Mfullmass.GetRows(); i++) {
140  result(this->offset + i) += c_a * Mfullmass(i, i);
141  }
142 }

144 // Build the mass matrix (for these variables) scaled by c_a, storing
145 // it in 'storage' sparse matrix, at given column/row offset.
146 // Note, most iterative solvers don't need to know mass matrix explicitly.
147 // Optimized: doesn't fill unneeded elements except mass and 3x3 inertia.
148 void ChVariablesBodyAddedMass::Build_M(ChSparseMatrix& storage, int insrow, int inscol, const double c_a) {
149  for (int row = 0; row < Mfullmass.GetRows(); ++row)
150  for (int col = 0; col < Mfullmass.GetColumns(); ++col)
151  storage.SetElement(insrow + row, inscol + col, c_a * Mfullmass.GetElement(row, col));
152 }
