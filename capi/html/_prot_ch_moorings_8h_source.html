853 void cppCable::setDragForce() {
854  /*
855  * setFluidVelocityAtNodes and setFluidDensityAtNodes
856  * must be called before this function
857  */
858  ChVector<> u_ch; // velocity from chrono
859  ChVector<> u_prot; // velocity from proteus
860  ChVector<> u_rel; // relative velocity of node with surrounding fluid
861  ChVector<> t_dir; // tangent at node
862  ChVector<> Fd_a; // axial (tangential) drag force
863  ChVector<> Fd_n; // normal(transversal) drag force
864  ChVector<> Fd; // total drag force
865  ChVector<> Va;
866  ChVector<> Vn;
867  double rho_f;
868  // clear current drag forces
869  double length_elem = length / (nb_nodes - 1);
870  for (int i = 0; i < nb_nodes; ++i) {
871  if (beam_type == "CableANCF") {
872  t_dir = nodes[i]->GetD();
873  u_ch = nodes[i]->GetPos_dt();
874  }
875  else if (beam_type == "BeamEuler") {
876  t_dir = nodesRot[i]->GetRot().GetVector();
877  u_ch = nodesRot[i]->GetPos_dt();
878  }
879  // get velocity u_prot from proteus // TO CHANGE !!
880  double ux_prot = fluid_velocity[i][0];
881  double uy_prot = fluid_velocity[i][1];
882  double uz_prot = fluid_velocity[i][2];
883  u_prot = ChVector<>(ux_prot, uy_prot, uz_prot);
884  u_rel = u_prot - u_ch;
885  // CAREFUL HERE: ChBeamElementANCF, GetD() does not give direction but normal
886  rho_f = fluid_density[i];
887  double dot = u_rel^t_dir;
888  Va = t_dir*dot;
889  Vn = u_rel-Va;
890  Fd_a = 0.5*rho_f*Cd_axial*M_PI*d*Va.Length()*Va;//(force per unit length)
891  Fd_n = 0.5*rho_f*Cd_normal*d*Vn.Length()*Vn;//(force per unit length)
892  Fd = Fd_a + Fd_n;
893  forces_drag[i]->Set(Fd);
894  }
895 }

898 void cppCable::setAddedMassForce() {
899  /*
900  * setFluidVelocityAtNodes and setFluidDensityAtNodes
901  * must be called before this function
902  */
903  ChVector<> a_ch; // acceleration from chrono
904  ChVector<> a_prot; // acceleration from proteus
905  ChVector<> a_rel; // relative acceleration of node with surrounding fluid
906  ChVector<> t_dir; // tangent at node
907  ChVector<> Fm_a; // axial (tangential) added mass force
908  ChVector<> Fm_n; // normal(transversal) added mass force
909  ChVector<> Fm_f; // fluid part added mass force
910  ChVector<> Fm; // total added mass force
911  ChVector<> Va;
912  ChVector<> Vn;
913  double rho_f;
914  // clear current drag forces
915  double length_elem = length / (nb_nodes - 1);
916  for (int i = 0; i < nb_nodes; ++i) {
917  if (beam_type == "CableANCF") {
918  t_dir = nodes[i]->GetD();
919  a_ch = nodes[i]->GetPos_dtdt();
920  }
921  else if (beam_type == "BeamEuler") {
922  t_dir = nodesRot[i]->GetRot().GetVector();
923  a_ch = nodesRot[i]->GetPos_dtdt();
924  }
925  // get velocity u_prot from proteus // TO CHANGE !!
926  double ax_prot = fluid_acceleration[i][0];
927  double ay_prot = fluid_acceleration[i][1];
928  double az_prot = fluid_acceleration[i][2];
929  a_prot = ChVector<>(ax_prot, ay_prot, az_prot);
930  a_rel = a_prot - a_ch;
931  rho_f = fluid_density[i];
932  double dot = a_rel^t_dir;
933  Va = t_dir*dot;
934  Vn = a_rel-Va;
935  Fm_a = rho_f*Cm_axial*M_PI*d*d/4.*Va;//(force per unit length)
936  Fm_n = rho_f*Cm_normal*M_PI*d*d/4.*Vn;//(force per unit length)
937  Fm_f = rho_f*M_PI*d*d/4.*a_prot;
938  Fm = Fm_a + Fm_n + Fm_f;
939  forces_addedmass[i]->Set(Fm);
940  }

943 void cppCable::applyForces() {

945  ChVector<> Fa = ChVector<>(0.,0.,0.);

949  Fb = Fb+*forces_drag[i+1].get();

952  Fa = Fa+*forces_addedmass[i].get();
953  Fb = Fb+*forces_addedmass[i+1].get();

960  }
961  else {
