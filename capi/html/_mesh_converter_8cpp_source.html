699 /* algorithm courtesy of Sebastian Rettenberger:
700  use brokers/routers for the vertex global ids.
701  Although we have used this trick before (see mpas/apfMPAS.cc),
702  I didn't think to use it here, so credit is given. */
703 static void constructResidence(Mesh2* m, GlobalToVert& globalToVert)
704 {
705  Gid max = getMax(globalToVert);
706  Gid total = max + 1;
707  int peers = PCU_Comm_Peers();
708  int quotient = total / peers;
709  int remainder = total % peers;
710  int mySize = quotient;
711  int self = PCU_Comm_Self();
712  if (self == (peers - 1))
713  mySize += remainder;
714  typedef std::vector< std::vector<int> > TmpParts;
715  TmpParts tmpParts(mySize);
716  /* if we have a vertex, send its global id to the
717  broker for that global id */
718  PCU_Comm_Begin();
719  APF_ITERATE(GlobalToVert, globalToVert, it) {
720  int gid = it->first;
721  int to = std::min(peers - 1, gid / quotient);
722  PCU_COMM_PACK(to, gid);
723  }
724  PCU_Comm_Send();
725  int myOffset = self * quotient;
726  /* brokers store all the part ids that sent messages
727  for each global id */
728  while (PCU_Comm_Receive()) {
729  int gid;
730  PCU_COMM_UNPACK(gid);
731  int from = PCU_Comm_Sender();
732  tmpParts.at(gid - myOffset).push_back(from);
733  }
734  /* for each global id, send all associated part ids
735  to all associated parts */
736  PCU_Comm_Begin();
737  for (int i = 0; i < mySize; ++i) {
738  std::vector<int>& parts = tmpParts[i];
739  for (size_t j = 0; j < parts.size(); ++j) {
740  int to = parts[j];
741  int gid = i + myOffset;
742  int nparts = parts.size();
743  PCU_COMM_PACK(to, gid);
744  PCU_COMM_PACK(to, nparts);
745  for (size_t k = 0; k < parts.size(); ++k)
746  PCU_COMM_PACK(to, parts[k]);
747  }
748  }
749  PCU_Comm_Send();
750  /* receiving a global id and associated parts,
751  lookup the vertex and classify it on the partition
752  model entity for that set of parts */
753  while (PCU_Comm_Receive()) {
754  int gid;
755  PCU_COMM_UNPACK(gid);
756  int nparts;
757  PCU_COMM_UNPACK(nparts);
758  Parts residence;
759  for (int i = 0; i < nparts; ++i) {
760  int part;
761  PCU_COMM_UNPACK(part);
762  residence.insert(part);
763  }
764  MeshEntity* vert = globalToVert[gid];
765  m->setResidence(vert, residence);
766  }
767 }

769 /* given correct residence from the above algorithm,
770  negotiate remote copies by exchanging (gid,pointer)
771  pairs with parts in the residence of the vertex */
772 static void constructRemotes(Mesh2* m, GlobalToVert& globalToVert)
773 {
774  int self = PCU_Comm_Self();
775  PCU_Comm_Begin();
776  APF_ITERATE(GlobalToVert, globalToVert, it) {
777  int gid = it->first;
778  MeshEntity* vert = it->second;
779  Parts residence;
780  m->getResidence(vert, residence);
781  APF_ITERATE(Parts, residence, rit)
782  if (*rit != self) {
783  PCU_COMM_PACK(*rit, gid);
784  PCU_COMM_PACK(*rit, vert);
785  }
786  }
787  PCU_Comm_Send();
788  while (PCU_Comm_Receive()) {
789  int gid;
790  PCU_COMM_UNPACK(gid);
791  MeshEntity* remote;
792  PCU_COMM_UNPACK(remote);
793  int from = PCU_Comm_Sender();
794  MeshEntity* vert = globalToVert[gid];
795  m->addRemote(vert, from, remote);
796  }
797 }
