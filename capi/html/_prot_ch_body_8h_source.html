528 void cppRigidBody::getTriangleMeshSDF(ChVector<> pos,
529  double* dist_n) {

531  auto nns = trimesh->getCoordsNormals();
532  ChVector<> dist_vec;
533  double min_dist=1e10;
534  double dist;
535  for (int i = 0; i < xxs.size(); i++) {
536  dist_vec = pos-xxs[i];
537  dist = dist_vec.Length();
538  if (dist < min_dist) {
539  min_dist = dist;
540  }
541  if (dist_vec.Dot(nns[i]) > 0) { // outside
542  min_dist = min_dist;
543  }
544  else { // inside
545  min_dist = -min_dist;
546  }
547  }
548  dist_n[0] = min_dist;
