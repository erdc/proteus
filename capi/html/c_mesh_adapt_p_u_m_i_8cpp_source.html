656 double MeshAdaptPUMIDrvr::getTotalMass()
662 {
663  apf::Field* voff = m->findField("vof");
664  assert(voff);
665  apf::MeshEntity* e;
666  apf::MeshIterator* it = m->begin(m->getDimension());
667  double mass = 0.0;
668  while ((e = m->iterate(it))) {
669  apf::MeshElement* elem = apf::createMeshElement(m,e);
670  apf::Element* voff_elem = apf::createElement(voff, elem);
671  int int_order = 4;
672  for(int l = 0; l < apf::countIntPoints(elem, int_order); ++l) {
673  apf::Vector3 qpt;
674  apf::getIntPoint(elem,int_order,l,qpt);
675  double vof_val = apf::getScalar(voff_elem,qpt);
676  double rho_val = getMPvalue(vof_val,rho[0],rho[1]);
677  double weight = apf::getIntWeight(elem,int_order,l);
678  apf::Matrix3x3 J;
679  apf::getJacobian(elem,qpt,J); //evaluate the Jacobian at the quadrature point
680  double Jdet = apf::getJacobianDeterminant(J,m->getDimension());
681  mass += rho_val*weight*Jdet;
682  }
683  apf::destroyElement(voff_elem);
684  apf::destroyMeshElement(elem);
685  }
686  m->end(it);
