566 bool FMMEikonalSolverBase::solve(const double* phi0,
567  const double * nodalSpeeds,
568  double * T, double zeroTol,
569  double trialTol,
570  int initFlagIn,// -1 -->ignore, 0 --> magn. 1--> frontInt
571  int verbose)
572 {
573  /***********************************************************************
574  Test first order fast marching method algorithm for eikonal equation
575 
576  \|\grad T \| = 1, \phi(\vec x) = 0, x \in \Gamma
577 
578  assuming \phi_0 describes initial location of interface Gamma and
579  has reasonable values (absolute values) for T close to Gamma. Here
580  T can be interpreted as the travel time from Gamma.
581 
582  Right now assumes global node numbers <--> global dofs but this can be
583  fixed easily
584  Input
585 
586 
587  phi0: dof array from P1 C0 FiniteElementFunction holding initial condition
588 
589  T : dof array from P1 C0 FiniteElementFunction for solution
590 
591 
592  Output
593  T(\vec x_n) : travel time from initial front to node (\vec x_n)
594 
595  Internal data structures
596 
597  Status : status of nodal point (dictionary)
598  -1 --> Far
599  0 --> Trial
600  1 --> Known
601  Trial : nodal points adjacent to front tuples (index,val) stored in heap
602 
603  ***********************************************************************/
604  using namespace std;
605  bool failed = false;
606  int nWastedRevisit = 0;
607  //take opportunity to reset some flags?
608  if (initFlagIn == 0)//MAGNITUDE ONLY
609  { initFlag = MAGNITUDE;}
610  else if (initFlagIn == 1)
611  { initFlag = FRONT_AND_MAGNITUDE;}
612  //otherwise don't change value constructed with
613 
614  //check sizes
615  if (Status.size() != unsigned(mesh->nNodes_global))
616  {
617  //c0 p1 required
618  if (verbose > 2)
619  cout<<"FMM solve resizing Status from "<<Status.size()<<" to "<<unsigned(mesh->nNodes_global)
620  <<endl;
621  Status.resize(mesh->nNodes_global,FAR);
622  }
623  StupidHeap Trial;
624  //mwf hack
625  //std::cout<<"In FMMandFSW baseSolve calling init"<<std::endl;
626  failed = initializeKnownPoints(phi0,T,zeroTol,verbose);
627  if (failed)
628  {
629  cout<<"initialize failed quitting"<<endl;
630  return failed;
631  }
632  //mwf hack
633  //std::cout<<"In FMMandFSW baseSolve back from calling init"<<std::endl;
634  assert(Known.size() > 0);
635  //loop through known points and calculate travel time for their unKNOWN neighbors
636  for (vector<int>::iterator pI = Known.begin(); pI != Known.end(); pI++)
637  {
638  const int I = *pI;
639  const int startI = mesh->nodeElementOffsets[I];
640  int nElements_starI = mesh->nodeElementOffsets[I+1]-startI;
641  for (int eN_star = 0; eN_star < nElements_starI; eN_star++)
642  {
643  const int eN = mesh->nodeElementsArray[startI+eN_star];
644  for (int nN_local = 0; nN_local < mesh->nNodes_element; nN_local++)
645  {
646  const int N_0 = mesh->elementNodesArray[eN*mesh->nNodes_element+nN_local];
647  if (N_0 != I && Status[N_0] != KNOWN)
648  {
649  double T_N = UNINITIALIZED;
650  bool updateFailed = localUpdate(N_0,T_N,T,nodalSpeeds[N_0],verbose);
651  if (!updateFailed)
652  {
653  T[N_0] = T_N;
654  if (Status[N_0] == TRIAL)
655  Trial.updateNode(N_0,T[N_0]);
656  else
657  Trial.insert(N_0,T[N_0]);
658  Status[N_0] = TRIAL;
659  }
660  }//neighbor that's UNKNOWN
661  }//nodes on element
662  }//elements in node star
663  }//Known nodes
664 
665  while (!Trial.isEmpty())
666  {
667  StupidHeap::EntryType minNode = Trial.pop();
668  const int I = minNode.first; const double T_I = minNode.second;
669  if (verbose > 1)
670  {
671  cout<<"*** FMM Start Trial loop min Trial point = ("<<I<<","<<T_I<<") Status= "
672  <<Status[I]<<endl;
673  }
674  if (Status[I] != KNOWN)
675  {
676  Status[I] = KNOWN; Known.push_back(I);
677  T[I] = T_I;
678  }
679  else
680  {
681  nWastedRevisit++;
682  }
683  //now update travel times of I's unKNOWN neighbors
684  const int startI = mesh->nodeElementOffsets[I];
685  const int nElements_starI = mesh->nodeElementOffsets[I+1]-startI;
686  for (int eN_star = 0; eN_star < nElements_starI; eN_star++)
687  {
688  const int eN = mesh->nodeElementsArray[startI+eN_star];
689  for (int nN_local = 0; nN_local < mesh->nNodes_element; nN_local++)
690  {
691  const int N_0 = mesh->elementNodesArray[eN*mesh->nNodes_element+nN_local];
692  if (N_0 != I && Status[N_0] != KNOWN)
693  {
694  double T_N;
695  bool updateFailed = localUpdate(N_0,T_N,T,nodalSpeeds[N_0],verbose);
696  if (!updateFailed)
697  {
698  T[N_0] = T_N;
699  if (Status[N_0] == TRIAL)
700  Trial.updateNode(N_0,T[N_0]);
701  else
702  Trial.insert(N_0,T[N_0]);
703  Status[N_0] = TRIAL;
704  }
705  }//neighbor that's unKNOWN
706  }//local nodes on neighboring element
707  }//elements in node star
708  }//Trial empty
709  //sanity check
710  if (verbose > 0)
711  {
712  if (Known.size() != unsigned(mesh->nNodes_global))
713  {
714  cout<<"Problem at end of FMM 2d sovle Known.size() = "<<Known.size()
715  <<" nNodes_global= "<<mesh->nNodes_global<<endl;
716  }
717  for (unsigned int I = 0; I < Status.size(); I++)
718  {
719  if (Status[I] != KNOWN)
720  {
721  cout<<"Problem TRIAL or FAR point found in Status["<<I<<"]= "<<Status[I]<<endl;
722  }
723  }
724  cout<<"FMM nWastedRevisit= "<<nWastedRevisit<<endl;
725  }
726  return failed;
727 }
